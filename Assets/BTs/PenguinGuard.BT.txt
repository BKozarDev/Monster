tree("Root")
	fallback
		tree("StayAndGuard")
		tree("Chase")

tree("StayAndGuard")
	sequence
		not IsOnGuardPosition

tree("Chase")
	//When the enemy is visible, attack it,
	//Otherwise forget about it.
	repeat
		sequence
			fallback
				while
					sequence
						IsPlayerVisible
					repeat
						sequence
							IsPlayerHasEgg
							fallback
								while
									parallel
										repeat
											not IsPlayerNear
										GoToLastPlayersPosition					
									Attack
				//Otherwise forget about it.
				sequence
					GoToPosition
